68inline void md5_step(uint32_t *buffer, uint32_t *input)
69{
70 uint32_t AA = buffer[0];
71 uint32_t BB = buffer[1];
72 uint32_t CC = buffer[2];
73 uint32_t DD = buffer[3];
74
75 uint32_t E;
76
77 unsigned int j;
78
79 for (unsigned int i = 0; i < 64; ++i)
80 {
81 switch (i / 16)

84 E = F(BB, CC, DD);
85 j = i;

88 E = G(BB, CC, DD);
89 j = ((i * 5) + 1) % 16;
90 break;
91 case 2:
92 E = H(BB, CC, DD);
93 j = ((i * 3) + 5) % 16;
94 break;
95 default:
96 E = I(BB, CC, DD);
97 j = (i * 7) % 16;
98 break;

102 DD = CC;
103 CC = BB;
104 BB = BB + rotate_left(AA + E + K[i] + input[j], S[i]);

106 }

108 buffer[0] += AA;
109 buffer[1] += BB;
110 buffer[2] += CC;
111 buffer[3] += DD;

137inline void md5_finalize(MD5_CTXT *ctx)
138{
139 uint32_t input[16];
140 unsigned int offset = ctx->size % 64;
141 unsigned int padding_length = offset < 56 ? 56 - offset : (56 + 64) - offset;
142
143 md5_update(ctx, PADDING, padding_length);
144 ctx->size -= (uint64_t)padding_length;
145
146 for (unsigned int j = 0; j < 14; ++j)
147 {
148 input[j] = (uint32_t)(ctx->input[(j * 4) + 3]) << 24 | (uint32_t)(ctx->input[(j * 4) + 2]) << 16 |
149 (uint32_t)(ctx->input[(j * 4) + 1]) << 8 | (uint32_t)(ctx->input[(j * 4)]);
150 }
151 input[14] = (uint32_t)(ctx->size * 8);
152 input[15] = (uint32_t)((ctx->size * 8) >> 32);
153
154 md5_step(ctx->buffer, input);
155

165inline void md5_string(char *input, uint8_t *result)
166{
