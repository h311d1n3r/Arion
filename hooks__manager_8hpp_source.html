171 template <typename... HookParams> void trigger_arion_hook(ARION_HOOK_TYPE type, HookParams... params)
172 {
173 for (auto &hook : this->hooks)
174 {
175 if (hook.second->type == type)
176 {
177 HOOK_CALLBACK &callback = hook.second->param->callback;
178 std::shared_ptr<Arion> arion = hook.second->param->arion.lock();
179 if (!arion)
180 throw arion_exception::ExpiredWeakPtrException("Arion");
181 void *user_data = hook.second->param->user_data;
182
183 std::visit(
184 [&](auto &&cb) {
185 using CallbackType = std::decay_t<decltype(cb)>;
186 if constexpr (std::is_invocable_v<CallbackType, std::shared_ptr<Arion>, HookParams..., void *>)
187 cb(arion, params..., user_data);
188 else
189 throw arion_exception::WrongHookParamsException();
190 },
191 callback);
192 }
193 }
194 }
