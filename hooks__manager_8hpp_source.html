654 template <typename... HookParams> void trigger_arion_hook(ARION_HOOK_TYPE type, HookParams... params)
655 {
656 for (auto &hook : this->hooks)
657 {
658 if (hook.second->type == type)
659 {
660 HOOK_CALLBACK &callback = hook.second->param->callback;
661 std::shared_ptr<Arion> arion = hook.second->param->arion.lock();
662 if (!arion)
663 throw arion_exception::ExpiredWeakPtrException("Arion");
664 void *user_data = hook.second->param->user_data;
665
666 std::visit(
667 [&](auto &&cb) {
668 using CallbackType = std::decay_t<decltype(cb)>;
669 if constexpr (std::is_invocable_v<CallbackType, std::shared_ptr<Arion>, HookParams..., void *>)
670 cb(arion, params..., user_data);
671 else
672 throw arion_exception::WrongHookParamsException();
673 },
674 callback);
675 }
676 }
677 }
