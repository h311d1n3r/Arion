20class ArionException : public std::exception
21{
22 private:
23 std::string message;
24
25 public:
26 ArionException(std::string msg) : message(msg)
27 {
28 void *buffer[EXCEPTION_MAX_FRAMES];
29 int frame_count = backtrace(buffer, EXCEPTION_MAX_FRAMES);
30 std::ostringstream ss;
31 ss << msg;
32 ss << std::endl << std::endl << "========== BACKTRACE ==========" << std::endl << std::endl;
33 for (int i = 0; i < frame_count; ++i)
34 {
35 Dl_info info;
36 if (dladdr(buffer[i], &info))
37 {
38 uintptr_t offset = 0;
39 if (info.dli_saddr)
40 offset = reinterpret_cast<uintptr_t>(buffer[i]) - reinterpret_cast<uintptr_t>(info.dli_saddr);
41 ss << std::setw(3) << i << ": "
42 << (info.dli_sname ? arion::demangle_cxx_symbol(std::string(info.dli_sname)) : "??") << " + "
43 << arion::int_to_hex<uintptr_t>(offset) << " (" << buffer[i] << ") at "
44 << (info.dli_fname ? info.dli_fname : "??") << std::endl;
45 }
46 else
47 {
48 ss << std::setw(3) << i << ": [unknown] (" << buffer[i] << ")\n";
49 }
50 }
51 ss << std::endl << "===============================" << std::endl << std::endl;
52 this->message = ss.str();
53 }
54
55 const char *what() const noexcept override
56 {
57 return message.c_str();
58 }
59};
