21class ArionException : public std::exception
22{
23 private:
25 std::string message;
26
27 public:
28 /*
29 * Builder for ArionException instances.
30 * @param[in] msg The reason why this exception was triggered.
31 */
32 ArionException(std::string msg) : message(msg)
33 {
34 void *buffer[ARION_EXCEPTION_MAX_FRAMES];
35 int frame_count = backtrace(buffer, ARION_EXCEPTION_MAX_FRAMES);
36 std::ostringstream ss;
37 ss << msg;
38 ss << std::endl << std::endl << "========== BACKTRACE ==========" << std::endl << std::endl;
39 for (int i = 0; i < frame_count; ++i)
40 {
41 Dl_info info;
42 if (dladdr(buffer[i], &info))
43 {
44 uintptr_t offset = 0;
45 if (info.dli_saddr)
46 offset = reinterpret_cast<uintptr_t>(buffer[i]) - reinterpret_cast<uintptr_t>(info.dli_saddr);
47 ss << std::setw(3) << i << ": "
48 << (info.dli_sname ? arion::demangle_cxx_symbol(std::string(info.dli_sname)) : "??") << " + "
49 << arion::int_to_hex<uintptr_t>(offset) << " (" << buffer[i] << ") at "
50 << (info.dli_fname ? info.dli_fname : "??") << std::endl;
51 }
52 else
53 {
54 ss << std::setw(3) << i << ": [unknown] (" << buffer[i] << ")\n";
55 }
56 }
57 ss << std::endl << "===============================" << std::endl << std::endl;
58 this->message = ss.str();
59 }
60
65 const char *what() const noexcept override
66 {
67 return message.c_str();
68 }
69};
