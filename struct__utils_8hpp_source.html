93 public:
94 ~PolymorphicStruct()

99 void set_field(std::string name, uint64_t val)
100 {
101 this->int_vals[name] = val;

103 void set_field(std::string name, void *arr)
104 {
105 auto arr_it = this->arr_vals.find(name);
106 if (arr_it != this->arr_vals.end())

108 this->arr_vals[name] = arr;

114 }

116 {
117 auto int_it = this->int_vals.find(name);
118 if (int_it == this->int_vals.end())

120 return int_it->second;
121 }
122 bool has_arr_field(std::string name)
123 {
124 auto arr_it = this->arr_vals.find(name);
125 return arr_it != this->arr_vals.end();
126 }
127 void *get_arr_field(std::string name)
128 {
129 auto arr_it = this->arr_vals.find(name);

422 arion::BYTE *build(STRUCT_ID id, arion::CPU_ARCH arch, size_t &struct_len)
423 {

425 if (struct_it == this->curr_structs.end())
426 throw arion_exception::WrongStructIdException();

428
429 std::vector<arion::BYTE> data_vec;
430 uint8_t align = this->get_align(arch);
431 uint8_t ptr_sz = this->get_ptr_sz(arch);
432 uint16_t arch_sz = this->get_arch_sz(arch);
433
434 for (POLYMORPHIC_STRUCT_FIELD field : this->fields)
435 {
436 if (field.constraint.arch != arion::CPU_ARCH::UNKNOWN_ARCH && arch != field.constraint.arch)
437 continue;
438 if (field.constraint.arch_sz != 0 && arch_sz != field.constraint.arch_sz)
439 continue;
440
441 switch (field.type)
442 {
443 case V8: {
444 uint8_t val = 0;
445 if (curr_struct->has_int_field(field.name))
446 val = curr_struct->get_int_field(field.name);
447 data_vec.push_back(val);
448 break;
449 }
450 case V16: {
451 if (data_vec.size() % sizeof(uint16_t))
452 {
453 uint8_t align_sz = sizeof(uint16_t) - (data_vec.size() % sizeof(uint16_t));
454 data_vec.insert(data_vec.end(), align_sz, 0);

457 if (curr_struct->has_int_field(field.name))
458 val = curr_struct->get_int_field(field.name);

460 break;
461 }

629 if (field.constraint.arch_sz != 0 && arch_sz != field.constraint.arch_sz)
630 continue;
631
632 if (ss.str().size() > 1)
633 ss << ", ";
634 ss << field.name << "=";
635 switch (field.type)
636 {
637 case V8:
638 case V16:
639 case V32:
640 case V64:
641 case PTR_SZ: {
642 uint64_t val = 0;
643 if (curr_struct->has_int_field(field.name))
644 val = curr_struct->get_int_field(field.name);
645 ss << field.arion_type->str(arion, val);
646 break;
647 }
648 case A8: {
649 ss << "{";
650 if (curr_struct->has_arr_field(field.name))
651 {
652 uint8_t *arr = (uint8_t *)curr_struct->get_arr_field(field.name);
653 for (size_t arr_i = 0; arr_i < field.sz; arr_i++)
654 {
655 ss << field.arion_type->str(arion, arr[arr_i]);
656 if (arr_i < field.sz - 1)
657 ss << ", ";
658 }
659 }
660 ss << "}";
661 break;
662 }
663 case A16: {
664 ss << "{";
665 if (curr_struct->has_arr_field(field.name))
666 {
667 uint16_t *arr = (uint16_t *)curr_struct->get_arr_field(field.name);
668 for (size_t arr_i = 0; arr_i < field.sz; arr_i++)
669 {
670 ss << field.arion_type->str(arion, arr[arr_i]);
671 if (arr_i < field.sz - 1)
672 ss << ", ";
673 }
674 }
675 ss << "}";
676 break;
677 }
678 case A32: {
679 ss << "{";
680 if (curr_struct->has_arr_field(field.name))
681 {
682 uint32_t *arr = (uint32_t *)curr_struct->get_arr_field(field.name);
683 for (size_t arr_i = 0; arr_i < field.sz; arr_i++)
684 {
685 ss << field.arion_type->str(arion, arr[arr_i]);
686 if (arr_i < field.sz - 1)
687 ss << ", ";
688 }
689 }
690 ss << "}";
691 break;
692 }
693 case A64: {
694 ss << "{";
695 if (curr_struct->has_arr_field(field.name))
696 {
697 uint64_t *arr = (uint64_t *)curr_struct->get_arr_field(field.name);
698 for (size_t arr_i = 0; arr_i < field.sz; arr_i++)
699 {
700 ss << field.arion_type->str(arion, arr[arr_i]);
701 if (arr_i < field.sz - 1)
702 ss << ", ";
703 }
704 }
705 ss << "}";
706 break;
707 }
708 case PTR_ARR: {
709 ss << "{";
710 if (curr_struct->has_arr_field(field.name))
711 {
712 void *arr = curr_struct->get_arr_field(field.name);
713 for (size_t arr_i = 0; arr_i < field.sz; arr_i++)
714 {
715 uint64_t val = 0;
716 memcpy(&val, (void *)((uint64_t)arr + arr_i * arch_sz), arch_sz);
717 ss << field.arion_type->str(arion, val);
718 if (arr_i < field.sz - 1)
719 ss << ", ";
720 }
721 }
722 ss << "}";
723 break;
724 }
725 default:
726 ss << "UNK_FIELD";
727 break;
728 }
729 }
730
731 ss << "}";
732
733 return ss.str();
734 }
735
736 void release_struct(STRUCT_ID id)
737 {
738 auto struct_it = this->curr_structs.find(id);
739 if (struct_it == this->curr_structs.end())
740 throw arion_exception::WrongStructIdException();
741
742 this->curr_structs.erase(id);
743 if (this->curr_structs.size())
744 this->free_ids.push(id);
745 else
746 {
747 this->free_ids = std::stack<STRUCT_ID>();
748 this->curr_id = 1;
749 }
750 }
751
752 void clear_structs()
753 {
754 std::vector<STRUCT_ID> struct_ids; // need to clone keys to prevent concurrency editing
755 for (const auto &struct_pair : this->curr_structs)
756 struct_ids.push_back(struct_pair.first);
757 for (STRUCT_ID struct_id : struct_ids)
758 this->release_struct(struct_id);
759 }

772{
773 private:
774 std::shared_ptr<arion_poly_struct::PolymorphicStructFactory<T>> factory;
775
776 protected:
777 ArionStructType(std::string name, std::shared_ptr<arion_poly_struct::PolymorphicStructFactory<T>> factory)
778 : arion_type::KernelType(name, arion::LOG_COLOR::BLUE), factory(factory) {};
779
780 public:
781 std::string str(std::shared_ptr<arion::Arion> arion, uint64_t val) override
782 {
783 if (!val || !this->arion_is_mapped(arion, val))
784 return arion::int_to_hex<uint64_t>(val);
785 arion::CPU_ARCH arch = this->arion_curr_arch(arion);
786 size_t struct_sz = this->factory->get_struct_sz(arch);
787 arion::BYTE *struct_buf = (arion::BYTE *)malloc(struct_sz);
788 std::vector<arion::BYTE> struct_data = this->arion_read_mem(arion, val, struct_sz);
789 memcpy(struct_buf, struct_data.data(), struct_sz);
790 arion_poly_struct::STRUCT_ID struct_id = this->factory->feed(arch, struct_buf);
791 std::string struct_str = this->factory->to_string(struct_id, arion, arch);
792 free(struct_buf);
793 this->factory->release_struct(struct_id);
794 return struct_str;
795 }
